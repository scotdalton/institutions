def includes_ip?(candidate)
  return false if ip_addresses.nil?
  return ip_addresses.include? candidate
end

def ip_addresses=(args)
  args.collect! do |arg|
    arg.match(%r-/) ? convert_to_range(arg) : arg
  end
  @ip_addresses = IPAddrRangeSet.new(*args)
end
